PROGRAM-03 : Code and analyze to find an optimal solution to Travelling Salesman Problem.
#include<iostream>
using namespace std;
int ary[10][10],completed[10],n,cost=0,v;
int least(int c)
{
	int i,nc=999;
	int min=999,kmin;
	for(i=1;i<=n;i++)
	{
		if((ary[c][i]!=1000)&&(completed[i]==0))
			if(ary[c][i]+ary[i][c] < min)
			{
				min=ary[i][c]+ary[c][i];
				kmin=ary[c][i];
				nc=i;
			}
	}
	if(min!=999)
	cost=cost+kmin;
	return nc;
}
void mincost(int city)
{
	int i,ncity;
	completed[city]=1;
	cout<<city<<"--->";
	ncity=least(city);
	if(ncity==999)
	{
		ncity=v;
		cout<<v;
		cost=cost+ary[city][v];
		return;
	}
	mincost(ncity);
}
int main()
{
	int i,edges,source,desti,w,j;
	cout<<" Enter the number of cites: ";
	cin>>n;
	cout<<"\n Enter the number of paths : ";
	cin>>edges;
	for(i=1;i<=n;i++)
	for(j=1;i<=n;i++)
	ary[i][j]=1000;
	for(i=1;i<=n;i++)
	completed[i]=0;
	for(i=1;i<=edges;i++)
	{
cout<<" Enter source of "<<i<<" path : ";
cin>>source;
cout<<" Enter destination of "<<i<<" path : ";
cin>>desti;
cout<<" Enter cost of "<<i<<" path : ";
cin>>w;
ary[source][desti]=w;
	}
	cout<<"\n Enter the starting city:";
	cin>>v;
	cout<<"\n\n The Path is:\n";
	mincost(v);
	cout<<"\n\n Minimum cost is "<<cost;
	return 0;
}
